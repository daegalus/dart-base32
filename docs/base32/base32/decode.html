  if (base32.isEmpty) {
    return Uint8List(0);
  }

  base32 = _pad(base32, encoding: encoding);

  if (!isValid(base32, encoding: encoding)) {
    throw FormatException('Invalid Base32 characters');
  }

  if (encoding == Encoding.crockford) {
    base32 = base32.replaceAll('-', '');

  var length = base32.indexOf('=');
